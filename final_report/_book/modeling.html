4.1 Data Loading

train <- read.csv("train.csv")
test <- read.csv("test.csv")

train <- train %>% 
  select(-X,-weekday) %>% 
  mutate_if(is.character,as.factor)

test <- test %>% 
  select(-X,-weekday) %>% 
  mutate_if(is.character,as.factor)
rec <- train %>% 
  recipe(occupancy ~.) %>% 
  step_normalize(all_numeric_predictors()) %>% 
  step_dummy(all_factor_predictors())
